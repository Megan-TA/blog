1. 两个有序数组合并成一个有序数组

```javascript
var a = [1, 7, 10];
var b = [2, 4, 12, 16];

function merge(arg1, arg2) {
  var m = arg1.length;
  var n = arg2.length;

  while (n > 0) {
    // 原数组的数比插入的数组数大 则将最大的数往后移动到 n+m-1位置
    // 对应的数组索引减一，继续和两一个数组的值比较
    if (arg1[m - 1] > arg2[n - 1]) {
      arg1[n + m - 1] = arg1[m - 1];
      m--;
    } else {
      arg1[n + m - 1] = arg2[n - 1];
      n--;
    }
  }

  return arg1;
}
```

2. 斐波那契数列实现

斐波那契数列指的是这样一个数列：1、1、2、3、5、8、13、21、34、……在数学上，斐波那契数列以如下被以递推的方法定义：F(1)=1，F(2)=1, F(n)=F(n-1)+F(n-2)（n>=3，n∈N\*）

```javascript
function fib(nMonth) {
  return nMonth <= 2
    ? 1
    : arguments.callee(nMonth - 1) + arguments.callee(nMonth - 2);
}
```

3. 冒泡排序

1、比较相邻的两个元素，如果前一个比后一个大，则交换位置。
2、比较完第一轮的时候，最后一个元素是最大的元素。
3、这时候最后一个元素是最大的，所以最后一个元素就不需要参与比较大小。

```javascript
function pop(arr) {
  for (let i = 0; i <= arr.length - 1; i++) {
    for (let j = 0; j <= arr.length - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        let temp = arr[j + 1];
        arr[j + 1] = arr[j];
        arr[j] = temp;
      }
    }
  }
  return arr;
}
```

4. 快速排序

```javascript
function quick(arr) {
  if (arr.length === 0) return arr;
  let p = arr[0];
  let leftArr = [];
  let rightArr = [];
  for (let i = 1; i <= arr.length - 2; i++) {
    if (arr[i] > p) {
      rightArr.push(arr[i]);
    } else {
      leftArr.push(arr[i]);
    }
  }

  return quick(leftArr).concat(p, quick(rightArr));
}
```
